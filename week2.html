Week 2 - 流程控制

學習目標：掌握條件判斷與迴圈結構，實現程式的流程控制

本週學習內容

  • 條件判斷：if、else if、else
  • 關係運算符與邏輯運算符
  • 多重選擇：switch-case
  • for 迴圈
  • while 與 do-while 迴圈
  • break 與 continue 敘述
  • 巢狀迴圈應用

1. 條件判斷（if-else）

基本 if 敘述

#include <stdio.h> int main() { int score = 85; // 單一條件判斷 if (score >= 60) { printf("及格！\n"); } return 0; }

if-else 雙向判斷

#include <stdio.h> int main() { int age; printf("請輸入年齡："); scanf("%d", &age); if (age >= 18) { printf("已成年，可以投票\n"); } else { printf("未成年，尚不能投票\n"); } return 0; }

多重條件判斷（else if）

#include <stdio.h> int main() { int score; printf("請輸入分數："); scanf("%d", &score); if (score >= 90) { printf("等級：A（優秀）\n"); } else if (score >= 80) { printf("等級：B（良好）\n"); } else if (score >= 70) { printf("等級：C（中等）\n"); } else if (score >= 60) { printf("等級：D（及格）\n"); } else { printf("等級：F（不及格）\n"); } return 0; }
重要觀念：
  • 條件判斷從上到下依序檢查
  • 找到第一個為真的條件就執行，然後跳出
  • 如果只有一行敘述，可以省略大括號，但建議保留
  • else 必須跟在 if 或 else if 後面

2. 關係與邏輯運算符

關係運算符

運算符 說明 範例 結果
== 等於 5 == 5 1 (真)
!= 不等於 5 != 3 1 (真)
> 大於 5 > 3 1 (真)
< 小於 5 < 3 0 (假)
>= 大於等於 5 >= 5 1 (真)
<= 小於等於 5 <= 3 0 (假)

邏輯運算符

#include <stdio.h> int main() { int age = 20; int score = 85; // && (AND)：兩個條件都要為真 if (age >= 18 && score >= 60) { printf("成年且及格\n"); } // || (OR)：至少一個條件為真 if (score >= 90 || age >= 18) { printf("成績優秀或已成年\n"); } // ! (NOT)：反轉條件 if (!(score < 60)) { printf("不是不及格（即及格）\n"); } return 0; }

短路求值

#include <stdio.h> int main() { int x = 5, y = 10; // && 短路：第一個為假，就不檢查第二個 if (x > 10 && y++ > 5) { printf("這不會執行\n"); } printf("y = %d\n", y); // y 還是 10，沒有遞增 // || 短路：第一個為真，就不檢查第二個 if (x < 10 || y++ > 5) { printf("這會執行\n"); } printf("y = %d\n", y); // y 還是 10 return 0; }
常見錯誤：
  • 使用 = 而非 == 進行比較
  • 條件表達式中不小心產生副作用（如 x++）
  • 複雜條件沒有適當使用括號

3. switch-case 多重選擇

基本語法

#include <stdio.h> int main() { int choice; printf("==== 選單 ====\n"); printf("1. 新增\n"); printf("2. 刪除\n"); printf("3. 修改\n"); printf("4. 查詢\n"); printf("請選擇："); scanf("%d", &choice); switch (choice) { case 1: printf("執行新增操作\n"); break; case 2: printf("執行刪除操作\n"); break; case 3: printf("執行修改操作\n"); break; case 4: printf("執行查詢操作\n"); break; default: printf("無效的選擇！\n"); } return 0; }

Fall-through 特性

#include <stdio.h> int main() { char grade; printf("請輸入等級 (A-F)："); scanf(" %c", &grade); switch (grade) { case 'A': case 'a': printf("優秀\n"); break; case 'B': case 'b': printf("良好\n"); break; case 'C': case 'c': printf("中等\n"); break; case 'D': case 'd': printf("及格\n"); break; case 'F': case 'f': printf("不及格\n"); break; default: printf("無效的等級\n"); } return 0; }
switch-case 要點：
  • 每個 case 後面通常要加 break，否則會繼續執行下一個 case
  • case 的值必須是常數或字面值
  • default 是可選的，處理其他情況
  • 多個 case 可以共用同一段程式碼（fall-through）

4. for 迴圈

基本語法

#include <stdio.h> int main() { // 輸出 1 到 10 for (int i = 1; i <= 10; i++) { printf("%d ", i); } printf("\n"); // 計算 1 到 100 的總和 int sum = 0; for (int i = 1; i <= 100; i++) { sum += i; } printf("1到100的總和：%d\n", sum); return 0; }

巢狀迴圈 - 九九乘法表

#include <stdio.h> int main() { printf("==== 九九乘法表 ====\n"); for (int i = 1; i <= 9; i++) { for (int j = 1; j <= 9; j++) { printf("%d*%d=%2d ", i, j, i*j); } printf("\n"); } return 0; }

5. while 與 do-while 迴圈

while 迴圈

#include <stdio.h> int main() { int i = 1; int sum = 0; while (i <= 10) { sum += i; i++; } printf("總和：%d\n", sum); return 0; }

do-while 迴圈

#include <stdio.h> int main() { int number; do { printf("請輸入1-10之間的數字："); scanf("%d", &number); if (number < 1 || number > 10) { printf("輸入錯誤，請重新輸入！\n"); } } while (number < 1 || number > 10); printf("你輸入的是：%d\n", number); return 0; }

6. break 與 continue

break - 跳出迴圈

#include <stdio.h> int main() { // 找到第一個能被7整除的數 for (int i = 1; i <= 100; i++) { if (i % 7 == 0) { printf("找到了：%d\n", i); break; } } return 0; }

continue - 跳過本次迭代

#include <stdio.h> int main() { // 輸出1到10中的奇數 for (int i = 1; i <= 10; i++) { if (i % 2 == 0) { continue; // 跳過偶數 } printf("%d ", i); } printf("\n"); return 0; }

本週練習題

基礎練習

  1. 奇偶數判斷
    撰寫程式，讀取一個整數，判斷並輸出它是奇數還是偶數。
  2. 最大值與最小值
    讀取三個整數，找出並輸出其中的最大值和最小值。
  3. 閏年判斷
    讀取年份，判斷是否為閏年。
    提示：能被4整除但不能被100整除，或能被400整除的年份是閏年。

迴圈練習

  1. 階乘計算
    讀取一個正整數 n，計算並輸出 n! 的值。
  2. 質數判斷
    撰寫程式，輸出 1 到 100 之間的所有質數。
  3. 費氏數列
    輸出費氏數列的前 20 項（1, 1, 2, 3, 5, 8, 13...）

進階練習

  1. 猜數字遊戲
    程式產生一個 1-100 的隨機數，使用者猜數字，
    程式提示「太大」或「太小」，直到猜中為止，並顯示猜測次數。
  2. 圖案列印
    使用巢狀迴圈列印以下圖案：
    *
**
***
****
*****
  3. 數字金字塔
    列印數字金字塔：
        1
   121
  12321
 1234321
123454321
  4. 計算機選單
    使用 switch-case 製作一個選單式計算機，
    讓使用者選擇運算類型後輸入兩個數字進行計算。
練習提示：
  • 先在紙上畫出流程圖
  • 從簡單的測試資料開始驗證
  • 注意迴圈的終止條件
  • 巢狀迴圈要仔細思考內外層的關係
