#ifndef STUDENT_H
#define STUDENT_H

#define MAX_NAME 50
#define MAX_SUBJECTS 5

// 學生結構
typedef struct {
    int id;                          // 學號
    char name[MAX_NAME];             // 姓名
    int age;                         // 年齡
    char gender;                     // 性別 ('M' or 'F')
    float scores[MAX_SUBJECTS];      // 各科成績
    float average;                   // 平均成績
} Student;

// 學生列表節點
typedef struct StudentNode {
    Student data;
    struct StudentNode *next;
} StudentNode;

// 學生管理系統
typedef struct {
    StudentNode *head;
    int count;
} StudentSystem;

// 函數宣告
void initSystem(StudentSystem *sys);
int addStudent(StudentSystem *sys, Student *student);
StudentNode* findStudentById(StudentSystem *sys, int id);
StudentNode* findStudentByName(StudentSystem *sys, const char *name);
int updateStudent(StudentSystem *sys, int id, Student *newData);
int deleteStudent(StudentSystem *sys, int id);
void displayStudent(Student *student);
void displayAllStudents(StudentSystem *sys);
void calculateAverage(Student *student);
void sortStudentsByScore(StudentSystem *sys);
void generateReport(StudentSystem *sys);
void freeSystem(StudentSystem *sys);

#endif

2.2 學生操作函數實作（student.c）

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "student.h"

// 初始化系統
void initSystem(StudentSystem *sys) {
    sys->head = NULL;
    sys->count = 0;
}

// 計算平均成績
void calculateAverage(Student *student) {
    float sum = 0;
    for (int i = 0; i < MAX_SUBJECTS; i++) {
        sum += student->scores[i];
    }
    student->average = sum / MAX_SUBJECTS;
}

// 新增學生
int addStudent(StudentSystem *sys, Student *student) {
    // 檢查學號是否重複
    if (findStudentById(sys, student->id) != NULL) {
        printf("錯誤：學號 %d 已存在！\n", student->id);
        return 0;
    }
    
    // 計算平均成績
    calculateAverage(student);
    
    // 建立新節點
    StudentNode *newNode = (StudentNode*)malloc(sizeof(StudentNode));
    if (newNode == NULL) {
        printf("錯誤：記憶體配置失敗！\n");
        return 0;
    }
    
    newNode->data = *student;
    newNode->next = sys->head;
    sys->head = newNode;
    sys->count++;
    
    return 1;
}

// 根據學號查詢學生
StudentNode* findStudentById(StudentSystem *sys, int id) {
    StudentNode *current = sys->head;
    while (current != NULL) {
        if (current->data.id == id) {
            return current;
        }
        current = current->next;
    }
    return NULL;
}

// 根據姓名查詢學生
StudentNode* findStudentByName(StudentSystem *sys, const char *name) {
    StudentNode *current = sys->head;
    while (current != NULL) {
        if (strcmp(current->data.name, name) == 0) {
            return current;
        }
        current = current->next;
    }
    return NULL;
}

// 更新學生資料
int updateStudent(StudentSystem *sys, int id, Student *newData) {
    StudentNode *node = findStudentById(sys, id);
    if (node == NULL) {
        printf("錯誤：找不到學號 %d 的學生！\n", id);
        return 0;
    }
    
    // 保留原學號
    newData->id = id;
    calculateAverage(newData);
    node->data = *newData;
    
    return 1;
}

// 刪除學生
int deleteStudent(StudentSystem *sys, int id) {
    StudentNode *current = sys->head;
    StudentNode *prev = NULL;
    
    while (current != NULL) {
        if (current->data.id == id) {
            if (prev == NULL) {
                sys->head = current->next;
            } else {
                prev->next = current->next;
            }
            free(current);
            sys->count--;
            return 1;
        }
        prev = current;
        current = current->next;
    }
    
    printf("錯誤：找不到學號 %d 的學生！\n", id);
    return 0;
}

// 顯示單一學生資料
void displayStudent(Student *student) {
    printf("\n========================================\n");
    printf("學號：%d\n", student->id);
    printf("姓名：%s\n", student->name);
    printf("年齡：%d\n", student->age);
    printf("性別：%c\n", student->gender);
    printf("科目成績：\n");
    const char *subjects[] = {"國文", "英文", "數學", "自然", "社會"};
    for (int i = 0; i < MAX_SUBJECTS; i++) {
        printf("  %s：%.1f\n", subjects[i], student->scores[i]);
    }
    printf("平均成績：%.2f\n", student->average);
    printf("========================================\n");
}

// 顯示所有學生
void displayAllStudents(StudentSystem *sys) {
    if (sys->head == NULL) {
        printf("\n目前沒有學生資料。\n");
        return;
    }
    
    printf("\n共有 %d 位學生：\n", sys->count);
    printf("%-8s %-12s %-6s %-6s %-8s\n", 
           "學號", "姓名", "年齡", "性別", "平均");
    printf("================================================\n");
    
    StudentNode *current = sys->head;
    while (current != NULL) {
        printf("%-8d %-12s %-6d %-6c %-8.2f\n",
               current->data.id,
               current->data.name,
               current->data.age,
               current->data.gender,
               current->data.average);
        current = current->next;
    }
}

// 成績排序（氣泡排序）
void sortStudentsByScore(StudentSystem *sys) {
    if (sys->head == NULL || sys->head->next == NULL) {
        return;
    }
    
    int swapped;
    StudentNode *ptr1;
    StudentNode *lptr = NULL;
    
    do {
        swapped = 0;
        ptr1 = sys->head;
        
        while (ptr1->next != lptr) {
            if (ptr1->data.average < ptr1->next->data.average) {
                Student temp = ptr1->data;
                ptr1->data = ptr1->next->data;
                ptr1->next->data = temp;
                swapped = 1;
            }
            ptr1 = ptr1->next;
        }
        lptr = ptr1;
    } while (swapped);
}

// 產生報表
void generateReport(StudentSystem *sys) {
    if (sys->head == NULL) {
        printf("沒有資料可以產生報表。\n");
        return;
    }
    
    float totalAvg = 0;
    float maxAvg = sys->head->data.average;
    float minAvg = sys->head->data.average;
    Student *topStudent = &sys->head->data;
    
    StudentNode *current = sys->head;
    while (current != NULL) {
        totalAvg += current->data.average;
        
        if (current->data.average > maxAvg) {
            maxAvg = current->data.average;
            topStudent = ¤t->data;
        }
        
        if (current->data.average < minAvg) {
            minAvg = current->data.average;
        }
        
        current = current->next;
    }
    
    printf("\n========== 統計報表 ==========\n");
    printf("學生總數：%d\n", sys->count);
    printf("全班平均：%.2f\n", totalAvg / sys->count);
    printf("最高平均：%.2f（%s）\n", maxAvg, topStudent->name);
    printf("最低平均：%.2f\n", minAvg);
    printf("==============================\n");
}

// 釋放系統資源
void freeSystem(StudentSystem *sys) {
    StudentNode *current = sys->head;
    while (current != NULL) {
        StudentNode *temp = current;
        current = current->next;
        free(temp);
    }
    sys->head = NULL;
    sys->count = 0;
}

// 儲存到二進位檔案
int saveToFile(StudentSystem *sys, const char *filename) {
    FILE *fp = fopen(filename, "wb");
    if (fp == NULL) {
        printf("錯誤：無法開啟檔案 '%s'\n", filename);
        return 0;
    }
    
    // 寫入學生數量
    fwrite(&sys->count, sizeof(int), 1, fp);
    
    // 寫入所有學生資料
    StudentNode *current = sys->head;
    while (current != NULL) {
        fwrite(¤t->data, sizeof(Student), 1, fp);
        current = current->next;
    }
    
    fclose(fp);
    printf("資料已儲存到 %s（共 %d 筆）\n", filename, sys->count);
    return 1;
}

// 從二進位檔案載入
int loadFromFile(StudentSystem *sys, const char *filename) {
    FILE *fp = fopen(filename, "rb");
    if (fp == NULL) {
        printf("提示：找不到資料檔案 '%s'，將建立新檔案。\n", filename);
        return 0;
    }
    
    // 讀取學生數量
    int count;
    fread(&count, sizeof(int), 1, fp);
    
    // 讀取所有學生資料
    for (int i = 0; i < count; i++) {
        Student student;
        if (fread(&student, sizeof(Student), 1, fp) == 1) {
            addStudent(sys, &student);
        }
    }
    
    fclose(fp);
    printf("已載入 %d 筆學生資料\n", sys->count);
    return 1;
}

// 匯出為 CSV
int exportToCSV(StudentSystem *sys, const char *filename) {
    FILE *fp = fopen(filename, "w");
    if (fp == NULL) {
        printf("錯誤：無法建立檔案 '%s'\n", filename);
        return 0;
    }
    
    // 寫入標題行
    fprintf(fp, "學號,姓名,年齡,性別,國文,英文,數學,自然,社會,平均\n");
    
    // 寫入資料
    StudentNode *current = sys->head;
    while (current != NULL) {
        fprintf(fp, "%d,%s,%d,%c,%.1f,%.1f,%.1f,%.1f,%.1f,%.2f\n",
                current->data.id,
                current->data.name,
                current->data.age,
                current->data.gender,
                current->data.scores[0],
                current->data.scores[1],
                current->data.scores[2],
                current->data.scores[3],
                current->data.scores[4],
                current->data.average);
        current = current->next;
    }
    
    fclose(fp);
    printf("資料已匯出到 %s\n", filename);
    return 1;
}

void clearInputBuffer() {
    int c;
    while ((c = getchar()) != '\n' && c != EOF);
}

void handleAddStudent(StudentSystem *sys) {
    Student student;
    
    printf("\n========== 新增學生 ==========\n");
    
    printf("學號：");
    scanf("%d", &student.id);
    clearInputBuffer();
    
    printf("姓名：");
    fgets(student.name, MAX_NAME, stdin);
    student.name[strcspn(student.name, "\n")] = 0;
    
    printf("年齡：");
    scanf("%d", &student.age);
    
    printf("性別 (M/F)：");

5. 主程式整合

5.1 主程式（main.c）

#include <stdio.h>
#include <stdlib.h>
#include "student.h"
#include "file_io.h"
#include "ui.h"

int main() {
    StudentSystem sys;
    initSystem(&sys);
    
    // 載入資料
    loadFromFile(&sys, DATA_FILE);
    
    int choice;
    int running = 1;
    
    while (running) {
        showMenu();
        scanf("%d", &choice);
        clearInputBuffer();
        
        switch (choice) {
            case 1:
                handleAddStudent(&sys);
                break;
            
            case 2:
                handleSearchStudent(&sys);
                break;
            
            case 3:
                handleUpdateStudent(&sys);
                break;
            
            case 4:
                handleDeleteStudent(&sys);
                break;
            
            case 5:
                displayAllStudents(&sys);
                break;
            
            case 6:
                sortStudentsByScore(&sys);
                printf("已依成績排序（高到低）\n");
                displayAllStudents(&sys);
                break;
            
            case 7:
                generateReport(&sys);
                break;
            
            case 8: {
                char filename[100];
                printf("請輸入檔案名稱（例如：students.csv）：");
                scanf("%s", filename);
                exportToCSV(&sys, filename);
                break;
            }
            
            case 0:
                printf("\n確定要離開嗎？(Y/N)：");
                char confirm;
                scanf(" %c", &confirm);
                if (confirm == 'Y' || confirm == 'y') {
                    saveToFile(&sys, DATA_FILE);
                    running = 0;
                }
                break;
            
            default:
                printf("無效的選擇，請重新輸入。\n");
        }
    }
    
    freeSystem(&sys);
    printf("\n系統已關閉，感謝使用！\n");
    
    return 0;
}
