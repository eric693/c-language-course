1. 遊戲專案：井字遊戲（Tic-Tac-Toe）

1.1 遊戲規則與設計

井字遊戲是一個簡單但完整的遊戲專案，可以展現所學的各種技巧。

1 | 2 | 3 ---+---+--- 4 | 5 | 6 ---+---+--- 7 | 8 | 9
專案功能：
  • 雙人對戰模式
  • 單人對電腦模式（AI）
  • 遊戲狀態顯示
  • 勝負判定
  • 遊戲記錄

1.2 完整實作

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define BOARD_SIZE 3

typedef struct {
    char board[BOARD_SIZE][BOARD_SIZE];
    char currentPlayer;
    int movesCount;
    int player1Wins;
    int player2Wins;
    int draws;
} Game;

// 初始化遊戲
void initGame(Game *game) {
    for (int i = 0; i < BOARD_SIZE; i++) {
        for (int j = 0; j < BOARD_SIZE; j++) {
            game->board[i][j] = ' ';
        }
    }
    game->currentPlayer = 'X';
    game->movesCount = 0;
}

// 顯示棋盤
void displayBoard(Game *game) {
    printf("\n");
    for (int i = 0; i < BOARD_SIZE; i++) {
        for (int j = 0; j < BOARD_SIZE; j++) {
            printf(" %c ", game->board[i][j]);
            if (j < BOARD_SIZE - 1) printf("|");
        }
        printf("\n");
        if (i < BOARD_SIZE - 1) {
            printf("---+---+---\n");
        }
    }
    printf("\n");
}

// 顯示位置編號
void displayPositions() {
    printf("\n位置編號：\n");
    printf(" 1 | 2 | 3 \n");
    printf("---+---+---\n");
    printf(" 4 | 5 | 6 \n");
    printf("---+---+---\n");
    printf(" 7 | 8 | 9 \n\n");
}

// 檢查位置是否有效且為空
int isValidMove(Game *game, int position) {
    if (position < 1 || position > 9) {
        return 0;
    }
    
    int row = (position - 1) / BOARD_SIZE;
    int col = (position - 1) % BOARD_SIZE;
    
    return game->board[row][col] == ' ';
}

// 執行移動
void makeMove(Game *game, int position) {
    int row = (position - 1) / BOARD_SIZE;
    int col = (position - 1) % BOARD_SIZE;
    
    game->board[row][col] = game->currentPlayer;
    game->movesCount++;
}

// 檢查獲勝
char checkWinner(Game *game) {
    // 檢查橫排
    for (int i = 0; i < BOARD_SIZE; i++) {
        if (game->board[i][0] != ' ' &&
            game->board[i][0] == game->board[i][1] &&
            game->board[i][1] == game->board[i][2]) {
            return game->board[i][0];
        }
    }
    
    // 檢查直排
    for (int j = 0; j < BOARD_SIZE; j++) {
        if (game->board[0][j] != ' ' &&
            game->board[0][j] == game->board[1][j] &&
            game->board[1][j] == game->board[2][j]) {
            return game->board[0][j];
        }
    }
    
    // 檢查對角線
    if (game->board[0][0] != ' ' &&
        game->board[0][0] == game->board[1][1] &&
        game->board[1][1] == game->board[2][2]) {
        return game->board[0][0];
    }
    
    if (game->board[0][2] != ' ' &&
        game->board[0][2] == game->board[1][1] &&
        game->board[1][1] == game->board[2][0]) {
        return game->board[0][2];
    }
    
    return ' ';
}

// 檢查是否平手
int isDraw(Game *game) {
    return game->movesCount == 9 && checkWinner(game) == ' ';
}

// AI 移動（簡單策略）
int getAIMove(Game *game) {
    // 策略1：檢查是否能獲勝
    for (int i = 1; i <= 9; i++) {
        if (isValidMove(game, i)) {
            int row = (i - 1) / BOARD_SIZE;
            int col = (i - 1) % BOARD_SIZE;
            game->board[row][col] = 'O';
            
            if (checkWinner(game) == 'O') {
                game->board[row][col] = ' ';
                return i;
            }
            game->board[row][col] = ' ';
        }
    }
    
    // 策略2：阻擋對手獲勝
    for (int i = 1; i <= 9; i++) {
        if (isValidMove(game, i)) {
            int row = (i - 1) / BOARD_SIZE;
            int col = (i - 1) % BOARD_SIZE;
            game->board[row][col] = 'X';
            
            if (checkWinner(game) == 'X') {
                game->board[row][col] = ' ';
                return i;
            }
            game->board[row][col] = ' ';
        }
    }
    
    // 策略3：佔據中心
    if (isValidMove(game, 5)) {
        return 5;
    }
    
    // 策略4：佔據角落
    int corners[] = {1, 3, 7, 9};
    for (int i = 0; i < 4; i++) {
        if (isValidMove(game, corners[i])) {
            return corners[i];
        }
    }
    
    // 策略5：隨機選擇
    for (int i = 1; i <= 9; i++) {
        if (isValidMove(game, i)) {
            return i;
        }
    }
    
    return -1;
}

// 切換玩家
void switchPlayer(Game *game) {
    game->currentPlayer = (game->currentPlayer == 'X') ? 'O' : 'X';
}

// 顯示統計
void displayStats(Game *game) {
    printf("\n========== 遊戲統計 ==========\n");
    printf("玩家 X 勝利：%d 次\n", game->player1Wins);
    printf("玩家 O 勝利：%d 次\n", game->player2Wins);
    printf("平手：%d 次\n", game->draws);
    printf("==============================\n");
}

// 雙人模式
void playTwoPlayers(Game *game) {
    char winner;
    int position;
    
    initGame(game);
    
    printf("\n========== 雙人對戰模式 ==========\n");
    
    while (1) {
        displayBoard(game);
        printf("玩家 %c 的回合\n", game->currentPlayer);
        displayPositions();
        printf("請選擇位置 (1-9)：");
        scanf("%d", &position);
        
        if (!isValidMove(game, position)) {
            printf("無效的移動！請重試。\n");
            continue;
        }
        
        makeMove(game, position);
        
        winner = checkWinner(game);
        if (winner != ' ') {
            displayBoard(game);
            printf("玩家 %c 獲勝！\n", winner);
            if (winner == 'X') game->player1Wins++;
            else game->player2Wins++;
            break;
        }
        
        if (isDraw(game)) {
            displayBoard(game);
            printf("平手！\n");
            game->draws++;
            break;
        }
        
        switchPlayer(game);
    }
}

// 單人模式（對電腦）
void playVsComputer(Game *game) {
    char winner;
    int position;
    
    initGame(game);
    
    printf("\n========== 單人對戰模式（對電腦）==========\n");
    printf("你是 X，電腦是 O\n");
    
    while (1) {
        displayBoard(game);
        
        if (game->currentPlayer == 'X') {
            printf("你的回合\n");
            displayPositions();
            printf("請選擇位置 (1-9)：");
            scanf("%d", &position);
            
            if (!isValidMove(game, position)) {
                printf("無效的移動！請重試。\n");
                continue;
            }
        } else {
            printf("電腦思考中...\n");
            position = getAIMove(game);
            printf("電腦選擇位置 %d\n", position);
        }
        
        makeMove(game, position);
        
        winner = checkWinner(game);
        if (winner != ' ') {
            displayBoard(game);
            if (winner == 'X') {
                printf("你贏了！\n");
                game->player1Wins++;
            } else {
                printf("電腦贏了！\n");
                game->player2Wins++;
            }
            break;
        }
        
        if (isDraw(game)) {
            displayBoard(game);
            printf("平手！\n");
            game->draws++;
            break;
        }
        
        switchPlayer(game);
    }
}

int main() {
    Game game = {{{' '}}, 'X', 0, 0, 0, 0};
    int choice;
    
    srand(time(NULL));
    
    printf("╔══════════════════════════════╗\n");
    printf("║     井字遊戲 Tic-Tac-Toe    ║\n");
    printf("╚══════════════════════════════╝\n");
    
    while (1) {
        printf("\n========== 主選單 ==========\n");
        printf("1. 雙人對戰\n");
        printf("2. 對戰電腦\n");
        printf("3. 查看統計\n");
        printf("0. 離開遊戲\n");
        printf("===========================\n");
        printf("請選擇：");
        scanf("%d", &choice);
        
        switch (choice) {
            case 1:
                playTwoPlayers(&game);
                break;
            case 2:
                playVsComputer(&game);
                break;
            case 3:
                displayStats(&game);
                break;
            case 0:
                printf("\n感謝遊玩！\n");
                displayStats(&game);
                return 0;
            default:
                printf("無效的選擇！\n");
        }
    }
    
    return 0;
}

2. 程式碼重構與優化

2.1 重構原則

重構的目標：
  • 提高可讀性：讓程式碼更容易理解
  • 改善結構：模組化、減少重複
  • 增強維護性：容易修改和擴充
  • 優化效能：提升執行效率

2.2 常見重構技巧

// 重構前：重複的程式碼
void processStudent1() {
    printf("處理學生資料\n");
    // 大量重複程式碼
    printf("儲存到資料庫\n");
}

void processStudent2() {
    printf("處理學生資料\n");
    // 大量重複程式碼
    printf("儲存到資料庫\n");
}

// 重構後：提取共用函數
void processStudentData(Student *s) {
    // 共用邏輯
}

void saveToDatabase(Student *s) {
    // 儲存邏輯
}

void processStudent(Student *s) {
    processStudentData(s);
    saveToDatabase(s);
}


// 重構前：Magic Numbers
if (score >= 90) {
    grade = 'A';
}

// 重構後：使用常數
#define GRADE_A_THRESHOLD 90

if (score >= GRADE_A_THRESHOLD) {
    grade = 'A';
}


// 重構前：過長的函數
void handleEverything() {
    // 100+ 行程式碼
}

// 重構後：分解成小函數
void validateInput() { }
void processData() { }
void generateOutput() { }

void handleEverything() {
    validateInput();
    processData();
    generateOutput();
}

2.3 效能優化技巧

// 優化 1：減少不必要的計算
// 優化前
for (int i = 0; i < strlen(str); i++) {  // 每次迴圈都計算長度
    // ...
}

// 優化後
int len = strlen(str);  // 只計算一次
for (int i = 0; i < len; i++) {
    // ...
}


// 優化 2：選擇適當的資料結構
// 優化前：使用陣列搜尋 - O(n)
int linearSearch(int arr[], int size, int target);

// 優化後：使用雜湊表或二分搜尋 - O(log n) 或 O(1)
int binarySearch(int arr[], int size, int target);


// 優化 3：避免記憶體洩漏
// 優化前
void processData() {
    int *data = malloc(100 * sizeof(int));
    // 處理資料
    // 忘記 free(data);
}

// 優化後
void processData() {
    int *data = malloc(100 * sizeof(int));
    if (data == NULL) return;
    
    // 處理資料
    
    free(data);  // 記得釋放
}

3. 除錯技巧

3.1 常見錯誤類型

編譯錯誤：
  • 語法錯誤
  • 型別不匹配
  • 未宣告的變數
  • 缺少標頭檔
執行時錯誤：
  • 除以零
  • 陣列越界
  • 空指標解參考
  • 記憶體洩漏
邏輯錯誤：
  • 演算法錯誤
  • 邊界條件處理不當
  • 迴圈條件錯誤
  • 運算優先順序問題

3.2 除錯方法

// 方法 1：使用 printf 除錯
void processArray(int arr[], int size) {
    printf("[DEBUG] 開始處理陣列，大小：%d\n", size);
    
    for (int i = 0; i < size; i++) {
        printf("[DEBUG] arr[%d] = %d\n", i, arr[i]);
        // 處理邏輯
    }
    
    printf("[DEBUG] 處理完成\n");
}

// 方法 2：使用 assert 檢查假設
#include <assert.h>

void divide(int a, int b) {
    assert(b != 0);  // 確保 b 不為零
    int result = a / b;
}

// 方法 3：使用條件編譯
#ifdef DEBUG
    #define LOG(msg) printf("[LOG] %s\n", msg)
#else
    #define LOG(msg)
#endif

void someFunction() {
    LOG("進入函數");
    // 函數邏輯
    LOG("離開函數");
}

3.3 記憶體除錯工具

valgrind --leak-check=full ./program

# 使用 GDB 除錯
gdb ./program

4.2 核心知識點總結

必須掌握的核心概念：
  • 指標：C 語言的靈魂，務必精通
  • 記憶體管理：malloc/free 的正確使用
  • 資料結構：鏈結串列、堆疊、佇列、樹
  • 檔案 I/O：文字檔與二進位檔處理
  • 演算法：排序、搜尋、遞迴、動態規劃
  • 模組化設計：標頭檔、函數分離、專案架構

4.3 最佳實務總結

程式設計最佳實務：
  1. 程式碼風格
    • 一致的命名規則
    • 適當的縮排與格式
    • 有意義的變數名稱
  2. 錯誤處理
    • 檢查所有返回值
    • 驗證輸入資料
    • 提供有用的錯誤訊息
  3. 記憶體管理
    • 配對使用 malloc/free
    • 避免記憶體洩漏
    • 檢查 NULL 指標
  4. 程式碼組織
    • 函數保持簡短
    • 減少全域變數
    • 使用標頭檔組織介面
  5. 文件與註解
    • 函數加上說明註解
    • 複雜邏輯加上解釋
    • 維護 README 文件
